The styled API for React-like frameworks, such as React, React Native, Preact...

## Installation

```bash npm2yarn
npm install @fower/styled
```

## API

```jsx
import { styled } from '@fower/styled'
```

## Usage

**Styling HTML element**:

```jsx live
function StyledElement() {
  const Text = styled('span')
  return (
    <Text text3XL fontBold green400>
      Hello world
    </Text>
  )
}
```

**Styling Component**:

```jsx live
function StyledAnyComponent() {
  const Button = ({ children, className }) => <button className={className}>{children}</button>
  const StyledButton = styled(Button)

  return (
    <StyledButton textLG white bgGreen400 p-10 rounded-4 border-0>
      styled button
    </StyledButton>
  )
}
```
