import Tabs from '@theme/Tabs'
import TabItem from '@theme/TabItem'

| Atomic Props    | CSS Properties         |
| --------------- | ---------------------- |
| block           | display: block;        |
| inline          | display: inline;       |
| inline-block    | display: inline-block; |
| flex            | display: flex;         |
| inline-flex     | display: inline-flex;  |
| grid            | display: grid;         |
| hidden          | display: none;         |
| display={value} | display: {value};      |

## Block

Use `block` props to create a block-level element.

<Tabs
defaultValue="React"
values={[
{ label: 'React', value: 'React', },
{ label: 'Vue', value: 'Vue', },
]
}>
<TabItem value="React">

```jsx live
<Box spaceY-10>
  <Box bgAmber400 rounded-8 h-60 block></Box>
  <Box bgAmber400 rounded-8 h-60 block></Box>
  <Box bgAmber400 rounded-8 h-60 block></Box>
</Box>
```

</TabItem>
<TabItem value="Vue">

```html
<div spaceY-10>
  <div bgAmber400 rounded-8 h-60 block></div>
  <div bgAmber400 rounded-8 h-60 block></div>
  <div bgAmber400 rounded-8 h-60 block></div>
</div>
```

</TabItem>
</Tabs>
