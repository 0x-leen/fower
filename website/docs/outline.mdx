import Tabs from '@theme/Tabs'
import TabItem from '@theme/TabItem'

Atomic props for controlling the stack order of an element.

| Atomic Props    | CSS Properties    |
| --------------- | ----------------- |
| outlineNone     | outline: none;    |
| outline={value} | outline: {value}; |

## None Outline

Use `outlineNone` props to remove outline of an element.

<Tabs
defaultValue="React"
values={[
{ label: 'React', value: 'React', },
{ label: 'Vue', value: 'Vue', },
]
}>
<TabItem value="React">

```jsx live
<Box toEvenly>
  <Input p2 bgGray200 placeholder="default input" />
  <Input p2 outlineNone bgGray200 placeholder="none outline" />
</Box>
```

</TabItem>
<TabItem value="Vue">

```html
<div toEvenly>
  <input p2 bgGray200 placeholder="default input" />
  <input p2 outlineNone bgGray200 placeholder="none outline" />
</div>
```

</TabItem>
</Tabs>

## Set Outline

Use `outline` props to set outline for an element.

<Tabs
defaultValue="React"
values={[
{ label: 'React', value: 'React', },
{ label: 'Vue', value: 'Vue', },
]
}>
<TabItem value="React">

```jsx live
<Box toEvenly>
  <Input p2 bgGray200 placeholder="default input" />
  <Input p2 bgGray200 outline--focus="2px dotted red" placeholder="outline on focus" />
</Box>
```

</TabItem>
<TabItem value="Vue">

```html
<div toEvenly>
  <input p2 bgGray200 placeholder="default input" />
  <input p2 bgGray200 outline--focus="2px dotted red" placeholder="outline on focus" />
</div>
```

</TabItem>
</Tabs>
