import Tabs from '@theme/Tabs'
import TabItem from '@theme/TabItem'

| Atomic Props | CSS Properties              |
| ------------ | --------------------------- |
| uppercase    | text-transform: uppercase;  |
| lowercase    | text-transform: lowercase;  |
| capitalize   | text-transform: capitalize; |
| normalcase   | text-transform: none;       |

## Usage

Use `normalcase`, `uppercase`, `lowercase`, `capitalize` to control the Text Transform.

<Tabs
defaultValue="React"
values={[
{ label: 'React', value: 'React', },
{ label: 'Vue', value: 'Vue', },
]
}>
<TabItem value="React">

```jsx live
<Box textXL>
  <Box uppercase>Lorem ipsum dolor sit amet</Box>
  <Box lowercase>Lorem ipsum dolor sit amet</Box>
  <Box capitalize>Lorem ipsum dolor sit amet</Box>
  <Box normalcase>Lorem ipsum dolor sit amet</Box>
</Box>
```

</TabItem>
<TabItem value="Vue">

```html
<div textXL>
  <div uppercase>Lorem ipsum dolor sit amet</div>
  <div lowercase>Lorem ipsum dolor sit amet</div>
  <div capitalize>Lorem ipsum dolor sit amet</div>
  <div normalcase>Lorem ipsum dolor sit amet</div>
</div>
```

</TabItem>
</Tabs>
