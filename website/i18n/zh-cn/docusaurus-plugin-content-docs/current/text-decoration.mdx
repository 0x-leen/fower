import Tabs from '@theme/Tabs'
import TabItem from '@theme/TabItem'

| Atomic Props | CSS Properties                 |
| ------------ | ------------------------------ |
| underline    | text-decoration: underline;    |
| lineThrough  | text-decoration: line-through; |
| noUnderline  | text-decoration: none;         |

## Usage

Use `underline`, `lineThrough`, `noUnderline` to control the Text Decoration.

<Tabs
defaultValue="React"
values={[
{ label: 'React', value: 'React', },
{ label: 'Vue', value: 'Vue', },
]
}>
<TabItem value="React">

```jsx live
<Box textXL>
  <Box underline>Lorem ipsum dolor sit amet</Box>
  <Box lineThrough>Lorem ipsum dolor sit amet</Box>
  <Box noUnderline>Lorem ipsum dolor sit amet</Box>
</Box>
```

</TabItem>
<TabItem value="Vue">

```html
<div textXL>
  <div underline>Lorem ipsum dolor sit amet</div>
  <div lineThrough>Lorem ipsum dolor sit amet</div>
  <div noUnderline>Lorem ipsum dolor sit amet</div>
</div>
```

</TabItem>
</Tabs>
